Arrays

Arrays

An array is a collection Object that has elements indexed. The elements can be accessed by calling their index. Arrays can hold values of any type, from Number to String to Function.

          
            var sports = [];
            sports[0] = 'running';
            sports[1] = 'cycling';
            sports[2] = 'trekking';

            console.log(sports); [ 'running', 'cycling', 'trekking' ]

Arrays have a variety of array methods:

  • .length lists number of elements in array.
  • .concat() joins one or more arrays into one array 
                  
                var flowers = ['Lilly', 'Rose', 'Tulip'];
                var colors = ['red', 'blue', 'yellow'];
                var colorFlowers = flowers.concat(colors);
                console.log(colorFlowers); [ 'Lilly', 'Rose', 'Tulip', 'red', 'blue', 'yellow' ]
  • .indexOf() returns the index of an item in the array
  • .lastIndexOf() searches from the end of array for the last item's index. 
                  
                var transport = ['car', 'bus', 'train', 'airplane', 'taxi', 'ferry', 'train'];
                console.log(
                  transport.indexOf('train'), 2
                  transport.lastIndexOf('train')  6
                );
  • .shift() removes and returns the first element of an array
  • .pop() removes and returns last element of array 
                  
                function firstitem(array) {
                  return array.shift();
                }

                function lastItem(array) {
                  return array.pop();
                }

                console.log(
                  firstitem(['car', 'bus', 'train', 'airplane', 'taxi', 'ferry']),
                  lastItem(['car', 'bus', 'train', 'airplane', 'taxi', 'ferry'])
                ); car ferry
  • .push() adds a new element to then end of array
  • .unshift() adds a new element at the beginning of array and returns new length. 
                  
                function addFirstItem(array, item) {
                  return array.unshift(item);
                }

                function addlastItem(array, item) {
                  return array.push(item);
                }

                var adventureSports = ['paragliding', 'kayaking'];
                addlastItem(adventureSports, 'snowboarding');
                addFirstItem(adventureSports, 'buggykiting');

                console.log(adventureSports); [ 'buggykiting', 'paragliding', 'kayaking', 'snowboarding' ]
  • .reverse() reverses the order of elements and returns the reversed array.
  • .slice() returns a section of the array
  • .sort() sorts the elements in the array and returns the sorted array
  • .splice() adds and removes elements from the array and returns the removed elements array
  • .toString() creates a string from the elements in an array
  • .valueOf() returns the primitive values of an array as a comma-delimited string.
  • .filter() filters an array and returns items that match.
  • .map() maps the elements of an array and returns a new array with the mapped data.
  • .reduce() reduces array elements a single value.
  • .forEach() iterates through the array elements.
